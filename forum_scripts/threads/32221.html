Yes.

Like you would assign a value to a normal input tag.

Code:
<form:hidden path="yourProperty"/>
of with the old spring tags

Code:
<spring:bind path="command.yourProperty">
    <input type="hidden" value="${status.value}" name="${status.expression}"/>
</spring:bind>
Also checkout the examples provided with spring and read the reference guide. They give very clear information how to set/bind to form input tags.

You can only bind to the formBackingObject, not just any random object in your model.

Also you have

Code:
<spring:bind path="recipientTO.groupName">
    <input type="hidden" name="groupName"/>
</spring:bind>
while my code looks like

Code:
<spring:bind path="command.yourProperty">
    <input type="hidden" value="${status.value}" name="${status.expression}"/>
</spring:bind>
You need to tell it what the name is and what the value is. Just including a input type hidden inside the spring:bind tags doesn't do anything.

However if you are able to use Spring 2.0 I strongly recommend that. It lets you use the new form tags library. Replacing the 3 lines above with 1 single line.

Code:
<form:hidden path="yourProperty"/>
Also have you checked the samples and the reference guide? They provide all the information needed to create a simple working sample.

Assuming all the selectboxes are on the same page, why not simple bind the select boxes to a property on the command object? Why the complicated work around with javascript and hidden values?

The jsp could look something like this (Spring 2.0 sample)

Code:
<%@taglib prefix="c" uri="http://java.sun.com/jsp/jstl/core"%>
<%@taglib prefix="fmt" uri="http://java.sun.com/jsp/jstl/fmt"%>
<%@taglib prefix="fn" uri="http://java.sun.com/jsp/jstl/functions"%>

<%@taglib prefix="spring" uri="http://www.springframework.org/tags"%>
<%@taglib prefix="form" uri="http://www.springframework.org/tags/form" %>

<html>
<head>

<script type="text/javascript">
	function submitForm(step) {
		myForm.step.value=step;
		myForm.submit();
	}
</script>

</head>
<body>

<form:form name="myForm">
	<input type="hidden" name="step" value=""/>
	<form:select path="property1" onchange="submitForm('step1');">
		<form:options items="${yourfirstcollectionofitems}"/>
	</form:select>

	<form:select path="property2" onchange="submitForm('step2');">
		<c:choose>
			<c:when test="${fn:length(yoursecondcollectionofitems) eq 0}">
				<form:option value="" label="Select value from first list."/>;
			</c:when>
			<c:otherwise>
				<form:options items="${yoursecondcollectionofitems}"/>
			</c:otherwise>
		</c:choose>
	</form:select>

	<form:select path="property3" onchange="submitForm('finish');">
		<c:choose>
			<c:when test="${fn:length(yoursecondcollectionofitems) eq 0}">
				<form:option value="" label="Select value from first list."/>;
			</c:when>
			<c:when test="${fn:length(yourthirdcollectionofitems) eq 0}">
				<form:option value="" label="Select value from second list."/>;
			</c:when>			
			<c:otherwise>
				<form:options items="${yourthirdcollectionofitems}"/>
			</c:otherwise>
		</c:choose>
	</form:select>
	
	
</form:form>
</body>

</html>
And the matching controller

Code:
import java.util.HashMap;
import java.util.Map;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import org.springframework.validation.BindException;
import org.springframework.validation.Errors;
import org.springframework.web.bind.ServletRequestUtils;
import org.springframework.web.servlet.ModelAndView;
import org.springframework.web.servlet.mvc.SimpleFormController;
import org.springframework.web.util.WebUtils;


public class SomeController extends SimpleFormController {

    public SomeController() {
        super();
        setSessionForm(true);
    }

    @Override
    protected Map referenceData(HttpServletRequest request, Object command, Errors errors) throws Exception {
        
        Map refData = new HashMap();
        String action = ServletRequestUtils.getStringParameter(request, "step");
        //Retrieve values for first list, put them in the reference data.
        
        if ("step2".equalsIgnoreCase(action)) {
            // retrieve third list with values and put them in reference data
        } 
        
        if ("step2".equalsIgnoreCase(action) || "step1".equalsIgnoreCase(action)){
            //Retrieve values for second list and put them in reference data
        }
        
        return refData;
    }
    
    @Override
    protected ModelAndView onSubmit(HttpServletRequest request, HttpServletResponse response, Object command, BindException errors) throws Exception {
        // When the seletions have finish (action=finish) do the real form submission
        return super.onSubmit(request, response, command, errors);
    }
    
    @Override
    protected boolean isFormSubmission(HttpServletRequest request) {
        //When the submitted action is 'finish' we must process the form.
        try {
            String action = ServletRequestUtils.getStringParameter(request, "step");
            return "finish".equalsIgnoreCase(action);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}
