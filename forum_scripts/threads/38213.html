      this.storeName = name;
   }

   public String getStoreName() {
      return storeName;
   }


   private void initBikes() {
      bikes.add(new Bike("Shimano", "Roadmaster", 20, "11111", 15,
            "Fair"));
      bikes.add(new Bike("Cannondale", "F2000 XTR", 18, "22222",12,
            "Excellent"));
      bikes.add(new Bike("Trek","6000", 19, "33333", 12.4,
            "Fair"));
   }

   public ArrayListRentABike() {
     initBikes();
   }

   public ArrayListRentABike(String storeName) {
      this.storeName = storeName;
      initBikes();
   }

   public String toString() { return "com.springbook.RentABike: " + storeName; }

   public List getBikes() { return bikes; }

   public Bike getBike(String serialNo) {
      Iterator iter = bikes.iterator();
      while(iter.hasNext()) {
         Bike bike = (Bike)iter.next();
         if(serialNo.equals(bike.getSerialNo())) return bike;

      }

      return null;
   }

   public void saveBike(Bike bike) {
      deleteIfContains(bike);
      bikes.add(bike);
   }

   public void deleteBike(Bike bike) {
      deleteIfContains(bike);
   }

   private void deleteIfContains(Bike bike) {
      Iterator iter = bikes.iterator();
      while(iter.hasNext()) {
         Bike comp = (Bike)iter.next();
         if(comp.getManufacturer().equals(bike.getManufacturer()) && comp.getModel().equals(bike.getModel())) {
            bikes.remove(comp);
            return;
         }
      }
   }
}

For your ModelAndView, return "bikes.bikes" as the view name. You'll need to set up a viewResolver in your Spring config. Generally you'll use something like this:
Code:
<bean id="viewResolver" class="org.springframework.web.servlet.view.InternalResourceViewResolver">
        <property name="viewClass" value="org.springframework.web.servlet.view.JstlView"/>
        <property name="prefix" value="/WEB-INF/view/"/>
        <property name="suffix" value=".jsp"/>
    </bean>
How have you set up your DispatcherServlet in your web.xml file? Did you assign the *.bikes URL pattern to the Spring dispatcher?

Also, did you try changing the web.xml file to reference the 2.3 DTD yet?

If things still aren't working, let's turn on log4j logging for Spring. Create a file called log4j.xml in your WEB-INF/classes directory. Paste the following in it:

Code:
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE log4j:configuration SYSTEM "log4j.dtd">

<log4j:configuration xmlns:log4j="http://jakarta.apache.org/log4j/" debug="true">

    <appender name="CONSOLE" class="org.apache.log4j.ConsoleAppender">
        <layout class="org.apache.log4j.PatternLayout">
            <param name="ConversionPattern"
                value="%-5p | %d{ISO8601} | %c | %m%n"/>
        </layout>
    </appender>
    
    <logger name="org.springframework">
        <level value="DEBUG"/>
    </logger>

</log4j:configuration>
You should see output in your console. Paste what shows up when you make the HTTP request. The .jsp.jsp message makes it sound like you're closer.

   private String storeName;
   final List bikes = new ArrayList();

   public void setStoreName(String name) {
      this.storeName = name;
   }

   public String getStoreName() {
      return storeName;
   }


   private void initBikes() {
      bikes.add(new Bike("Shimano", "Roadmaster", 20, "11111", 15,
            "Fair"));
      bikes.add(new Bike("Cannondale", "F2000 XTR", 18, "22222",12,
            "Excellent"));
      bikes.add(new Bike("Trek","6000", 19, "33333", 12.4,
            "Fair"));
   }

   public ArrayListRentABike() {
     initBikes();
   }

   public ArrayListRentABike(String storeName) {
      this.storeName = storeName;
      initBikes();
   }

   public String toString() { return "com.springbook.RentABike: " + storeName; }

   public List getBikes() { return bikes; }

   public Bike getBike(String serialNo) {
      Iterator iter = bikes.iterator();
      while(iter.hasNext()) {
         Bike bike = (Bike)iter.next();
         if(serialNo.equals(bike.getSerialNo())) return bike;

      }

      return null;
   }

   public void saveBike(Bike bike) {
      deleteIfContains(bike);
      bikes.add(bike);
   }

   public void deleteBike(Bike bike) {
      deleteIfContains(bike);
   }

   private void deleteIfContains(Bike bike) {
      Iterator iter = bikes.iterator();
      while(iter.hasNext()) {
         Bike comp = (Bike)iter.next();
         if(comp.getManufacturer().equals(bike.getManufacturer()) && comp.getModel().equals(bike.getModel())) {
            bikes.remove(comp);
            return;
         }
      }
   }
}

Actually, at this point you shouldn't need the c:out tags at all - with the 2.4 switch in your web.xml, you're running in JSP 2.0 mode, in which case the container is evaluating the EL expressions. Try just putting this in the body of the page and see what happens:
Code:
${21+21}
If "42" is printed, you don't need the c:out tag and everything is good. Also, since you're running in JSP 2.0 mode, you can/should remove the taglib declaration from your web.xml file - because the JSTL taglibs will be auto-discovered.

Try that and post back.
