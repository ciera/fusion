package com.ibm.pcs.core.web;

import java.lang.reflect.*;
import java.util.*;

import javax.servlet.http.*;

import org.springframework.validation.*;

import com.ibm.pcs.data.*;
import com.ibm.pcs.manageorder.web.ManageOrderForm;

/**
 * Tab Form Controller
 * 
 * @author vasu
 *
 */
public abstract class TabFormController extends PCSMultiActionFormController {
	
	public transient static String CopyRightNotice1 = "IBM Confidential OCO Source Material";
	public transient static String CopyRightNotice2 = "(C) Copyright IBM Corp. 2006, 2007  All Rights Reserved.";
	public transient static String CopyRightNotice3 = "The source code for this program is not published or otherwise divested of its trade";
	public transient static String CopyRightNotice4 = "secrets, irrespective of what has been deposited with the U. S. Copyright Office.";

	private TabList tabList;

	protected Map referenceData(HttpServletRequest request, Object command, Errors errors) throws Exception {
		log.print("TabFormController.referenceData");
		log.print("tabListSize", tabList.getTabs().size());

		Map model = new HashMap();
		model.put("tabList", tabList.getTabs());		
		
		model.put("lastTab", Integer.toString(tabList.getTabs().size()));		// @01
		
		
		//Call tab specific referenceData
		//Method shoud be of signature : referenceDataXXXTab
		TabCommand tabCommand = (TabCommand) command;
		String methodName = "referenceData" + tabCommand.getSelectedTab().getDescription() + "Tab";		
		Method method;
		try {
			log.print("Attempting to call referenceData method" , methodName);
			log.print("Reflection Method: " + this.getClass().getMethod("referenceData", new Class[] { HttpServletRequest.class, Object.class, org.springframework.validation.Errors.class}));
								
			method = this.getClass().getMethod(methodName, new Class[] { HttpServletRequest.class, getCommandClass(), org.springframework.validation.Errors.class, java.util.Map.class});
			List params = new ArrayList(4);
			params.add(request);
			params.add(command);
			params.add(errors);
			params.add(model);			
			method.invoke(this, params.toArray(new Object[params.size()]));
		} catch (NoSuchMethodException e) {
			log.error("No tab-specific referenceData method, no big deal", e);
		}
		return model;
	}

	public void setTabList(TabList list) {
		tabList = list;
	}

}
